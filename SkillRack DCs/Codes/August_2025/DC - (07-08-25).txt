char* asciiToString(int N , int *ASCII_values_array){
    
    char *string = ((char *) malloc(((N + 1) * sizeof(char))));
    
    for(int index = 0 ; (index < N) ; index++){
        sprintf(string , "%s%c" , string , ((char) ASCII_values_array[index]));
    }
    
    return string;
}
