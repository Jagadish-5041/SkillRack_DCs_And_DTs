Problem Statement :-

Print The Floating Point Values in DC (Daily Challenge) in SkillRack in Python - Programming

Code :-

def is_within_bounds(index , N):
    
    return (0 <= index < N)

S = input().strip()

N = len(S)

for dot_index in [index for index in range(0 , N) if(S[index] == '.')]:
    
    left_half_string = ""
    
    left_pointer = (dot_index - 1)
    
    while((is_within_bounds(left_pointer , N)) and (S[left_pointer].isdigit())):
        
        left_half_string = (S[left_pointer] + left_half_string)
        
        left_pointer = (left_pointer - 1)
    
    right_half_string = ""
    
    right_pointer = (dot_index + 1)
    
    while((is_within_bounds(right_pointer , N)) and (S[right_pointer].isdigit())):
        
        right_half_string = (right_half_string + S[right_pointer])
        
        right_pointer = (right_pointer + 1)
    
    (print(left_half_string , '.' , right_half_string , sep = "")) if(left_half_string and right_half_string) else ("")
