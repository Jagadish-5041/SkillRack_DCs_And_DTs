def is_primary_diagonal(current_i , current_j , R , C):
    
    return (current_i == current_j)

def is_secondary_diagonal(current_i , current_j , R , C):
    
    return ((current_i + current_j) == (C - 1))

N = int(input())

R = N

C = N

matrix = [list(map(int , input().split())) for i in range(0 , R)]
