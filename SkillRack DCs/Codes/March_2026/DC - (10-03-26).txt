def get_peaks_count(N , checkpoints_list):
    
    return sum(1 for index in range(1 , (N - 1)) if((checkpoints_list[index] > checkpoints_list[(index - 1)]) and (checkpoints_list[index] > checkpoints_list[(index + 1)])))
