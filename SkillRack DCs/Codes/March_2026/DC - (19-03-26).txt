def is_odd(N):
    
    return ((N & 1) == 1)

N = int(input())

odd_number_value = 1

even_number_value = ((N * N) - 1) if(is_odd((N * N))) else (N * N)

pattern_list = []

for position in range(1 , ((N * N) + 1)):
    
    if(is_odd(position)):
        pattern_list.append(odd_number_value)
        odd_number_value = (odd_number_value + 2)
    else:
        pattern_list.append(even_number_value)
        even_number_value = (even_number_value - 2)

[(print(value)) if(((index + 1) % N) == 0) else (print(value , end = " ")) for index , value in enumerate(pattern_list)]
