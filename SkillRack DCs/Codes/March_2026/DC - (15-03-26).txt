def get_layer_sum(starting_i , ending_i , starting_j , ending_j , R , C , matrix):
    
    layer_sum = 0
    
    for current_j in range(starting_j , (ending_j + 1)):
        
        layer_sum += matrix[starting_i][current_j]
    
    for current_i in range((starting_i + 1) , (ending_i + 1)):
        
        layer_sum += matrix[current_i][ending_j]
    
    for current_j in range((ending_j - 1) , (starting_j - 1) , -1):
        
        layer_sum += matrix[ending_i][current_j]
    
    for current_i in range((ending_i - 1) , starting_i , -1):
        
        layer_sum += matrix[current_i][starting_j]
    
    return layer_sum
