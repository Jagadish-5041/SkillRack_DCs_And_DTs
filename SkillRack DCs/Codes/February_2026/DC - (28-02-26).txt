import math

def get_length_of_N(N):
    
    return ((1) if(N == 0) else (int(math.log10(abs(N))) + 1))

def is_odd(N):
    
    return ((N & 1) == 1)

def is_even(N):
    
    return ((N & 1) == 0)

def compute_power_function_in_log_N_time_complexity(base_value , power_value):
    
    product_value = 1
    
    while(power_value > 0):
        
        product_value = (product_value * base_value) if(is_odd(power_value)) else (product_value)
        
        base_value = (base_value * base_value)
        
        power_value = (power_value // 2)
    
    return product_value

def depth_first_search(number , number_length , numbers_list):
    
    if(not is_even(number_length)):
        numbers_list.append(number)
        return 
    
    divisor = compute_power_function_in_log_N_time_complexity(10 , (number_length // 2))
    
    left_half = (number // divisor)
    
    right_half = (number % divisor)
    
    depth_first_search(left_half , get_length_of_N(left_half) , numbers_list)
    
    depth_first_search(right_half , get_length_of_N(right_half) , numbers_list)

N = int(input())
