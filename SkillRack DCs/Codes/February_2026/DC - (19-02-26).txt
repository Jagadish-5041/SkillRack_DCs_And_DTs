Problem Statement :-

Find Area - Radius (or) Diameter in DC (Daily Challenge) in SkillRack in Java - Programming

Code :-

import java.util.*;

class Circle {
    
    private double radius;
    
    public Circle(int radius){
        
        this.radius = radius;
    }
    
    public Circle(double diameter){
        
        this.radius = (diameter / 2.0);
    }
    
    public double getArea(){
        
        return ((22.0 / 7.0) * this.radius * this.radius);
    }
} // end of Circle class

public class Hello {
    
    public static void main(String[] args){
        
        Scanner sc = new Scanner(System.in);
        
        int Q = Integer.parseInt(sc.nextLine().trim());
        
        Circle circle = null;
        
        if(Q == 1){
            int radius = Integer.parseInt(sc.nextLine().trim());
            circle = new Circle(radius);
        }else if(Q == 2){
            double diameter = Double.parseDouble(sc.nextLine().trim());
            circle = new Circle(diameter);
        }
        
        System.out.print(String.format("%.2f" , circle.getArea()));
    }
} // end of Hello class
