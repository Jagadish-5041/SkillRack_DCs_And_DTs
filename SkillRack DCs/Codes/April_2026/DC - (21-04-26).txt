import sys

def depth_first_search_Jump_Game_II_LeetCode(current_index , N , values_list , dp_list):
    
    if(current_index == (N - 1)):
        return 0
    
    if(current_index >= N):
        return sys.maxsize
    
    if(dp_list[current_index] is not None):
        return dp_list[current_index]
    
    minimum_steps_count = sys.maxsize
    
    maximum_steps_allowed = (0) if(values_list[current_index] == 'B') else (2)
    
    for step in range(1 , (maximum_steps_allowed + 1)):
        
        current_steps_count = depth_first_search_Jump_Game_II_LeetCode((current_index + step) , N , values_list , dp_list)
        
        minimum_steps_count = (min(minimum_steps_count , (current_steps_count + 1))) if(current_steps_count != sys.maxsize) else (minimum_steps_count)
    
    dp_list[current_index] = minimum_steps_count
    
    return dp_list[current_index]
