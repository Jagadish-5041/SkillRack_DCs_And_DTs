Problem Statement :-

Find Minimum Billable Training Count in DC (Daily Challenge) in SkillRack in Python - Programming

Code :-

import sys
sys.setrecursionlimit(10**7)

n = int(input())
tree = [[] for _ in range(n)]
color = [0] * n
count = 0

for i in range(1, n):
    parent, val = map(int, input().split())
    tree[parent].append(i)
    color[i] = (val == 0)

def dfs(node):
    global count
    results = []

    for child in tree[node]:
        results.append(dfs(child))

    if any(r == 0 for r in results):
        count += 1
        return 2

    if any(r == 2 for r in results):
        return 1

    return 0 if color[node] else 1

if dfs(0) == 0:
    count += 1

print(count)
