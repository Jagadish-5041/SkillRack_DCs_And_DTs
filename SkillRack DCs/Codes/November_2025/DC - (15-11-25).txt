int is_within_bounds(int index , int N){
    
    return ((0 <= index) && (index < N));
}

}

    int *numbers_array = ((int *) malloc((N * sizeof(int))));

    char *operators_string = ((char *) malloc((N * sizeof(char))));

        int number = 0;

            number = ((number * 10) + ((int) (string[index++] - '0')));
