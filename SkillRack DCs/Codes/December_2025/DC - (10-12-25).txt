def get_digits_frequency_map(values_list):
    
    digits_frequency_map = {digit : 0 for digit in range(0 , 10)}
    
    for value in values_list:
        
        while(value > 0):
            
            digit = (value % 10)
            
            digits_frequency_map[digit] = (digits_frequency_map[digit] + 1)
            
            value = (value // 10)
    
    return digits_frequency_map
