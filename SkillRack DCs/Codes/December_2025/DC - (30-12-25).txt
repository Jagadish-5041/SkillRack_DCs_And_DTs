#include <string.h>

void findNextWordPosition(char *words_string){
    
    int N = strlen(words_string);
    
    char **words_array = ((char **) malloc((N * sizeof(char *))));
    
    for(int i = 0 ; (i < N) ; i++){
        words_array[i] = ((char *) malloc((N * sizeof(char))));
    }
    
    int words_array_index = 0;
    
    for(char *string_tokenizer = strtok(words_string , " ") ; (string_tokenizer != NULL) ; string_tokenizer = strtok(NULL , " ")){
        strcpy(words_array[words_array_index++] , string_tokenizer);
    }
    
    for(int i = 0 ; (i < words_array_index) ; i++){
        
        for(int j = ((i + 1) % words_array_index) ; ; j = ((j + 1) % words_array_index)){
            
            if(words_array[i][(strlen(words_array[i]) - 1)] == words_array[j][(strlen(words_array[j]) - 1)]){
                
                printf("%d " , (j + 1));
                
                break;
            }
        }
    }
}
