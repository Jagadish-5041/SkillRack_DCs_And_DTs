Rearrange Digits - Equal Count in DC (Daily Challenge) in SkillRack in Python - Programming

Code :-

import math

def get_length_of_N(N):
    
    return ((0) if(N == 0) else (int(math.log10(abs(N))) + 1))

N = int(input())

values_list = list(map(int , input().split()))

lengths_list = [get_length_of_N(value) for value in values_list]

maximum_frequency_length = (sorted(lengths_list)[1]) if(N > 2) else (sum(lengths_list) // 2)

index_1 = -999

index_2 = -999

for index in range(0 , N):
    
    index_1 = (index) if(maximum_frequency_length < lengths_list[index]) else (index_1)
    
    index_2 = (index) if(maximum_frequency_length > lengths_list[index]) else (index_2)

extra_digits_string = str(values_list[index_1])[maximum_frequency_length : ]

values_list[index_1] = int(str(values_list[index_1])[ : maximum_frequency_length])

values_list[index_2] = int((str(values_list[index_2]) + extra_digits_string))

print(*values_list)

print(sum(values_list))
