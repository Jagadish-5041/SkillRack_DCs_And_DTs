Problem Statement :-

Elevator Moving Pattern in DC (Daily Challenge) in SkillRack in Python - Programming

Code :-

def is_within_bounds(index , N):
    
    return (0 <= index < N)

N = int(input())

floor_values_list = list(map(int , input().split()))

R = 0

running_sum = 0

for floor_value in floor_values_list:
    
    running_sum += floor_value
    
    R = max(R , running_sum)

C = N

matrix = [['-' for j in range(0 , C)] for i in range(0 , R)]

current_i = (R - 1)

current_j = 0

while(is_within_bounds(current_j , C)):
    
    if(floor_values_list[current_j] > 0):
        
        for floor in range(1 , (floor_values_list[current_j] + 1)):
            
            matrix[current_i][current_j] = '*'
            
            current_i -= 1
    
    else:
        
        current_i += 1
        
        for floor in range(1 , (abs(floor_values_list[current_j]) + 1)):
            
            current_i += 1
            
            matrix[current_i][current_j] = '*'
        
        current_i -= 1
    
    current_j += 1

[print(*row_list , sep = "") for row_list in matrix]
