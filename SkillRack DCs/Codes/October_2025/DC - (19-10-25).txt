#include <string.h>

int is_within_bounds(int index , int N){
    
    return ((0 <= index) && (index < N));
}

char* getStringFromParentheses(char *string){
    
    int N = strlen(string);
    
    char *answer_string = ((char *) malloc((N * sizeof(char))));
    
    for(int index_1 = 0 ; (index_1 < N) ; index_1++){
        
        if(string[index_1] == '('){
            
            for(int index_2 = (index_1 + 1) ; ((is_within_bounds(index_2 , N)) && (string[index_2] != ')')) ; index_2 = ((index_2 + 1) % N)){
                
                strncat(answer_string , &string[index_2] , 1);
            }
        }
    }
    
    (strlen(answer_string) == 0) ? (sprintf(answer_string , "-1")) : (printf(""));
    
    return answer_string;
}
