Integer With Largest Digits Sum in DC (Daily Challenge) in SkillRack in Python - Programming

def get_digits_sum(N):
    
    digits_sum = 0
    
    while(N > 0):
        
        digit = (N % 10)
        
        digits_sum = (digits_sum + digit)
        
        N = (N // 10)
    
    return digits_sum

N = int(input().strip())

maximum_digits_sum = 0

data_list = []

for number in range(N , 0 , -1):
    
    digits_sum = get_digits_sum(number)
    
    maximum_digits_sum = max(maximum_digits_sum , digits_sum)
    
    data_list.append((number , digits_sum))

for number , digits_sum in data_list:
    
    if(digits_sum == maximum_digits_sum):
        
        print(number)
        
        quit()
