def depth_first_search(current_combination_string , visited_flag_list , digits_string , N):
    
    global all_combinations_set
    
    (all_combinations_set.add(int(current_combination_string))) if((current_combination_string) and (int(current_combination_string) != 0)) else ("")
    
    for index in range(0 , N):
        
        if(not visited_flag_list[index]):
            
            if((len(current_combination_string) == 0) and (digits_string[index] == 0)):
                continue
            
            visited_flag_list[index] = True
            
            depth_first_search((current_combination_string + digits_string[index]) , visited_flag_list , digits_string , N)
            
            visited_flag_list[index] = False
