def get_maximum_digit(N):
    
    maximum_digit = 0
    
    while(N > 0):
        
        current_digit = (N % 10)
        
        maximum_digit = max(maximum_digit , current_digit)
        
        N = (N // 10)
    
    return maximum_digit

def get_base_converted_value(N , target_base_value):
    
    new_N = 0
    
    power_value = 0
    
    while(N > 0):
        
        new_N += ((target_base_value ** power_value) * (N % 10))
        
        power_value += 1
        
        N //= 10
    
    return new_N
