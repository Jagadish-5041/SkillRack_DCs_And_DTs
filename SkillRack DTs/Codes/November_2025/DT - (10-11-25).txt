def process_list(values_list):
    
    processed_list = [values_list[0]]
    
    for index in range(1 , len(values_list)):
        
        if(not (values_list[index] > values_list[(index - 1)])):
            
            processed_list.append(values_list[index])
    
    return processed_list

N = int(input())

original_values_list = list(map(int , input().split()))

count = 0

while(True):
    
    processed_values_list = process_list(original_values_list)
    
    if((len(original_values_list) == 0) or (original_values_list == processed_values_list)):
        break
    
    original_values_list = processed_values_list
    
    count += 1

print(count)
