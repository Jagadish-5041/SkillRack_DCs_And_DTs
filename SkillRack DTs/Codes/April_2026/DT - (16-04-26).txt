Problem Statement :-

Maximum Of Odd And Even Digit Count in DT (Daily Test) in SkillRack in Python - Programming

Code :-

def is_odd(N):
    
    return ((N & 1) == 1)

def is_even(N):
    
    return ((N & 1) == 0)

def get_maximum_of_odd_and_even_digits_count(N):
    
    odd_digits_count = 0
    
    even_digits_count = 0
    
    while(N > 0):
        
        digit = (N % 10)
        
        odd_digits_count = (odd_digits_count + 1) if(is_odd(digit)) else (odd_digits_count)
        
        even_digits_count = (even_digits_count + 1) if(is_even(digit)) else (even_digits_count)
        
        N = (N // 10)
    
    return max(odd_digits_count , even_digits_count)

print(get_maximum_of_odd_and_even_digits_count(int(input())))
