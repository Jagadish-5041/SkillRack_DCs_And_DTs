def get_frequency_map(S):
    
    frequency_map = {alphabet : 0 for alphabet in "abcdefghijklmnopqrstuvwxyz"}
    
    [frequency_map.__setitem__(character , (frequency_map.get(character , 0) + 1)) for character in S]
    
    return frequency_map
