char *get_clean_string(char *string){
    
    char *copy_string = ((char *) malloc(((strlen(string) + 1) * sizeof(char))));
    
    int N = strlen(string);
    
    while((N >= 0) && ((string[(N - 1)] == '\r') || (string[(N - 1)] == '\n'))){
        N--;
    }
    
    sprintf(copy_string , "%.*s" , N , string);
    
    return copy_string;
}

int is_number(char *string){
    
    for(int N = strlen(string) , index = 0 ; (index < N) ; index++){
        
        if(!isdigit(string[index])){
            
            return 0;
        }
    }
    
    return 1;
}
