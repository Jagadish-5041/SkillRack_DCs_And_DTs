def get_clockwise_spirally_traversed_characters(starting_i , ending_i , starting_j , ending_j , R , C , matrix):
    
    characters_string = ""
    
    for current_j in range(starting_j , (ending_j + 1)):
        
        characters_string += matrix[starting_i][current_j]
    
    for current_i in range((starting_i + 1) , (ending_i + 1)):
        
        characters_string += matrix[current_i][ending_j]
    
    for current_j in range((ending_j - 1) , (starting_j - 1) , -1):
        
        characters_string += matrix[ending_i][current_j]
    
    for current_i in range((ending_i - 1) , starting_i , -1):
        
        characters_string += matrix[current_i][starting_j]
    
    return characters_string

def is_palindrome(N , string):
    
    for index in range(0 , ((N // 2) + 1)):
        
        if(string[index] != string[(N - index - 1)]):
            
            return False
    
    return True
