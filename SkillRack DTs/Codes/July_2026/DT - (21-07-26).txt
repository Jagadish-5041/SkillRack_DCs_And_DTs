#include <stdio.h>
#include <stdlib.h>
#include <limits.h>
#define jagadish_DT main

int is_even(int N){
    
    return ((N & 1) == 0);
}

int jagadish_DT(){
    
    int N;
    
    scanf("%d\n" , &N);
    
    int array[N];
    
    for(int index = 0 ; (index < N) ; index++){
        scanf("%d " , &array[index]);
    }
    
    for(int i = 0 ; (i < N) ; i++){
        
        int answer_value = array[i];
        
        for(int j = (i + 1) ; (j < N) ; j++){
            
            if((answer_value < array[j]) && (is_even(array[j]))){
                
                answer_value = array[j];
                
                break;
            }
        }
        
        printf("%d " , answer_value);
    }
    
    return 0;
}
