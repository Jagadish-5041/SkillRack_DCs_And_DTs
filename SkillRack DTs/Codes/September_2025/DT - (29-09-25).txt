def get_offset_value(character):
    
    return ((ord('a')) if(character.islower()) else (ord('A')))

def get_alphabet_position(alphabet):
    
    return (ord(alphabet) - get_offset_value(alphabet))

def get_next_alphabet(alphabet):
    
    return chr((((get_alphabet_position(alphabet) + 1) % 26) + get_offset_value(alphabet)))

def get_previous_alphabet(alphabet):
    
    return chr((((get_alphabet_position(alphabet) - 1) % 26) + get_offset_value(alphabet)))

R , C = map(int , input().split())

matrix = [list(map(int , input().split())) for i in range(0 , R)]

character = input().strip()

alphabet_1 = character

alphabet_2 = character

for i in range(0 , R):
    
    for j in range(0 , C):
        
        if(matrix[i][j] == 1):
            
            matrix[i][j] = alphabet_1
            
            alphabet_1 = get_next_alphabet(alphabet_1)
        
        else:
            
            matrix[i][j] = alphabet_2
            
            alphabet_2 = get_previous_alphabet(alphabet_2)

[print(*row_list) for row_list in matrix]
