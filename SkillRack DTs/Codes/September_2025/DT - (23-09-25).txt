def get_K_th_bit(N , K):
    
    return ((N >> (K - 1)) & 1)

N = int(input())

for index , number in enumerate(list(map(int , input().split()))):
    
    K += (get_K_th_bit(number , (index + 1)) * (2 ** powers_of_2))
    
    powers_of_2 += 1
