int get_digit(char digit_character){

}

long long int everyTwoDigitsToInteger(int N){
    
    long long int sum_value = 0;
    
    char *digits_string = ((char *) malloc((64 * sizeof(char))));
    
    sprintf(digits_string , "%d" , N);
    
    for(int M = strlen(digits_string) , index = 0 ; (index < (M - 1)) ; index++){
        
        long long int current_number = 0;
        
        for(int digit_1 = get_digit(digits_string[index]) , digit_2 = get_digit(digits_string[(index + 1)]) , current_digit = digit_1 , flag = (digit_1 <= digit_2) ; ((flag) ? (current_digit <= digit_2) : (current_digit >= digit_2)) ; ((flag) ? (current_digit++) : (current_digit--))){
            current_number = ((current_number * 10) + current_digit);
        }

    }
    
    free(digits_string);

}
