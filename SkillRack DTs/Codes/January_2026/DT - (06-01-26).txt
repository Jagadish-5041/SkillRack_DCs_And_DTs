Problem Statement :-

All Submatrices Sum - From Top-Left in DT (Daily Test) in SkillRack in Python - Programming

Code :-

R , C = map(int , input().split())

matrix = [list(map(int , input().split())) for i in range(0 , R)]

row_wise_prefix_sums_matrix = [[0 for j in range(0 , C)] for i in range(0 , R)]

for i in range(0 , R):
    
    current_row_running_sum = 0
    
    for j in range(0 , C):
        
        current_row_running_sum += matrix[i][j]
        
        row_wise_prefix_sums_matrix[i][j] = current_row_running_sum

column_wise_prefix_sums_matrix_for_row_wise_prefix_sums_matrix = [[0 for j in range(0 , C)] for i in range(0 , R)]

for j in range(0 , C):
    
    column_wise_prefix_sums_for_row_wise_prefix_sums_matrix = 0
    
    for i in range(0 , R):
        
        column_wise_prefix_sums_matrix_for_row_wise_prefix_sums_matrix[i][j] = column_wise_prefix_sums_for_row_wise_prefix_sums_matrix
        
        column_wise_prefix_sums_for_row_wise_prefix_sums_matrix += row_wise_prefix_sums_matrix[i][j]

for i in range(0 , R):
    
    for j in range(0 , C):
        
        print((row_wise_prefix_sums_matrix[i][j] + column_wise_prefix_sums_matrix_for_row_wise_prefix_sums_matrix[i][j]) , end = " ")
    
    print()
