def convert_minutes_to_seconds(minutes):
    
    return (minutes * 60)

def convert_total_seconds_to_1_hour_time_format(total_seconds):
    
    minutes = ((total_seconds // 60) % 60)
    
    seconds = (total_seconds - convert_minutes_to_seconds(minutes))
    
    return ("{:02d}:{:02d}".format(minutes , seconds))

def add_subtitle(time_string , subtitle_string , T , subtitles_map):
    
    minutes , seconds = map(int , time_string.split(":"))
    
    current_second = (convert_minutes_to_seconds(minutes) + seconds)
    
    for count in range(1 , (T + 1)):
        
        subtitles_map.setdefault(current_second , []).append(subtitle_string)
        
        current_second += 1
