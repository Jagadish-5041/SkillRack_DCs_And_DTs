def is_odd(N):
    
    return ((N & 1) == 1)

def depth_first_search(valid_strings_count , N , strings_and_positions_list , K):
    
    global answer_string
    
    if(valid_strings_count <= 0):
        return
    
    new_strings_and_positions_list = [('' , (index + 1)) for index in range(0 , N)]
    
    for string , position in strings_and_positions_list:
        
        if(len(string) < K):
            answer_string += string
            K += 1
        else:
            answer_string += ((string[ : K]) if(is_odd(position)) else (string[(K * -1) : ]))
            new_strings_and_positions_list[(position - 1)] = (((string[K : ]) if(is_odd(position)) else (string[ : (K * -1)])) , position)
    
    depth_first_search(sum(1 for string , position in new_strings_and_positions_list if(string)) , len(new_strings_and_positions_list) , new_strings_and_positions_list , K)
