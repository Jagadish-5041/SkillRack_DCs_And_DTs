int get_digit(char digit_character){
    
    return ((int) (digit_character - '0'));
}

int get_alphabet_index_in_Hexa_Decimal_format(char alphabet){
    
    return (((toupper(alphabet) - 'A')) + 10);
}

int get_integer_value(char character){
    
    return ((isalpha(character)) ? (get_alphabet_index_in_Hexa_Decimal_format(character)) : (get_digit(character)));
}

int is_odd(int N){
    
    return ((N & 1) == 1);
}

long long int compute_power_function(long long int base_value , long long int power_value){
    
    long long int product_value = 1;
    
    while(power_value > 0){
        
        product_value = (is_odd(power_value)) ? (product_value * base_value) : (product_value);
        
        base_value = (base_value * base_value);
        
        power_value = (power_value / 2);
    }
    
    return product_value;
}
